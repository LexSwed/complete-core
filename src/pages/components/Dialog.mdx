import { Playground } from 'dokz';
import { Dialog, Button, Heading, TextField, Flex, Box, Switch } from '@fxtrot/ui';

## Dialog

<Playground>
  <Box p="$8">
    <Dialog.Trigger>
      <Button>New workspace</Button>
      {(close) => (
        <Dialog.Modal>
          <Heading>Create new workspace</Heading>
          <Box
            pt="$4"
            as="form"
            onSubmit={(e) => {
              e.preventDefault();
              window.alert(e.currentTarget['workspace-name'].value);
            }}
          >
            <TextField placeholder="Carefive" name="workspace-name" label="Workspace name" />
            <Box pb="$10" />
            <Flex flow="row-reverse" space="sm">
              <Button type="submit">Create</Button>
              <Button onClick={close} variant="flat">
                Cancel
              </Button>
            </Flex>
          </Box>
        </Dialog.Modal>
      )}
    </Dialog.Trigger>
  </Box>
</Playground>
