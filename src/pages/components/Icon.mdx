import { Playground } from 'dokz';
import { Icon, Box, Text, TextField, Flex, Switch, styled } from '@fxtrot/ui';
import * as solidIcons from '@heroicons/react/solid';
import * as outlineIcons from '@heroicons/react/outline';

```tsx
import { Icon } from '@complete/core';
import { DotsHorizontalIcon } from '@heroicons/react/solid';
```

export const DotsHorizontalIcon = solidIcons.DotsHorizontalIcon;

<Playground>
  <Icon as={DotsHorizontalIcon} size="xl" />
</Playground>

export const Grid = styled('div', {
  display: 'grid',
  gridAutoFlow: 'row',
  gridTemplateColumns: 'repeat(auto-fill, minmax(150px, auto))',
  gap: '$2',
  py: '$4',
});

export const IconBox = styled(Flex, {
  'boxShadow': '$md',
  'px': '$3',
  'py': '$5',
  'transition': '0.2s ease-in-out',
  'cursor': 'pointer',
  'minWidth': 0,
  'overflow': 'hidden',
  '&:hover': {
    boxShadow: '$lg',
  },
});

export function copy(text) {
  navigator.permissions.query({ name: 'clipboard-write' }).then((result) => {
    if (result.state == 'granted' || result.state == 'prompt') {
      navigator.clipboard.writeText(text);
    }
  });
}

export const IconsWithFilter = () => {
  const [filterText, setText] = React.useState('');
  const [isOutline, setOutline] = React.useState(false);
  return (
    <Flex cross="stretch" gap="md">
      <Flex flow="row" gap="md" main="spread" cross="center">
        <TextField css={{ flexBasis: '50%' }} value={filterText} onChange={setText} placeholder="Search for an icon" />
        <Switch checked={isOutline} onChange={setOutline} label="Show outline icons" />
      </Flex>
      <Grid>
        {Object.entries(isOutline ? outlineIcons : solidIcons)
          .filter(([name]) => name.toLowerCase().includes(filterText.toLowerCase()))
          .map(([name, icon]) => (
            <IconBox
              gap="sm"
              cross="center"
              main="center"
              key={name}
              size="md"
              onClick={() => copy(`import { ${name} } from '@heroicons/react/${isOutline ? 'outline' : 'solid'}'`)}
            >
              <Icon as={icon} color="$gray700" size="xl" />
              <Text size="xs" ellipsis>
                {name}
              </Text>
            </IconBox>
          ))}
      </Grid>
    </Flex>
  );
};

<IconsWithFilter />
