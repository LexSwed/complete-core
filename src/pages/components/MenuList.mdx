<Playground>
  {() => {
    const [selected, setSelected] = React.useState('Appearance');
    return (
      <Box width="$56" p="$2">
        <MenuList label="Settings" role="navigation">
          {[
            {
              icon: HiOutlineUsers,
              label: 'Profiles',
            },
            {
              icon: HiOutlineAdjustments,
              label: 'Appearance',
            },
            {
              icon: HiOutlineGlobe,
              label: 'Language',
            },
          ].map(({ icon, label }) => (
            <MenuList.Item key={label} selected={selected === label} onClick={() => setSelected(label)}>
              <Icon as={icon} />
              <Text>{label}</Text>
            </MenuList.Item>
          ))}
        </MenuList>
      </Box>
    );
  }}
</Playground>
