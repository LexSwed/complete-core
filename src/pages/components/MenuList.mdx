import { Playground } from 'dokz';
import { Box, MenuList, ListItem, Icon, Text, Section, ThemeProvider } from '@fxtrot/ui';
import { HiOutlineUsers, HiOutlineAdjustments, HiOutlineGlobe } from 'react-icons/hi';

## MenuList

<Playground>
  {() => {
    const [selected, setSelected] = React.useState('Appearance');
    return (
      <Box width="$56" p="$4">
        <MenuList role="navigation">
          <Section title="Settings">
            {[
              {
                icon: HiOutlineUsers,
                label: 'Profiles',
              },
              {
                icon: HiOutlineAdjustments,
                label: 'Appearance',
              },
              {
                icon: HiOutlineGlobe,
                label: 'Language',
              },
            ].map(({ icon, label }) => (
              <MenuList.Item key={label} selected={selected === label} onClick={() => setSelected(label)}>
                <Icon as={icon} />
                <Text>{label}</Text>
              </MenuList.Item>
            ))}
          </Section>
        </MenuList>
      </Box>
    );
  }}
</Playground>
